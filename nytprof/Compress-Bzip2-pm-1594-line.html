467 if ( defined($buffer) ) {
468 @res = $obj->is_write ? $obj->bzdeflate( $buffer ) : $obj->bzinflate( $buffer );
469 return undef if $res[1] != &BZ_OK;
470
471 $out = $res[0];
472 }
473 $out = '' if !defined($out);
474
475 @res = $obj->bzclose;
476 return undef if $res[1] != &BZ_OK;
477
478 return $out.$res[0];
479}
